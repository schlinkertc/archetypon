SQLite Example

import pandas as pd
import os
import sqlalchemy
sqlalchemy.__version__
'1.4.46'
class SQLiteDB(Database):
    drivername='sqlite'
    database='test.db'

db = SQLiteDB()

users = pd.DataFrame({
    'id':[1,2,3],
    'user':['larry','moe','curly']
})

users.to_sql('users',db.engine,if_exists='replace',index=False)

queried = db.query_to_df("select * from users")
assert queried.equals(users)
os.remove('test.db')
